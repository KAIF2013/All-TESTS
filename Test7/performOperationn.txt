function performOperation(num1, num2, operator) {
  switch (operator) {
    case '+':
      return num1 + num2;
    case '-':
      return num1 - num2;
    case '*':
      return num1 * num2;
    case '/':
      if (num2 !== 0) {
        return num1 / num2;
      } else {
        return "Error: Division by zero";
      }
    default:
      return "Invalid operator";
  }
}


const number1 = 10;
const number2 = 5;
const operation = '+';  

const result = performOperation(number1, number2, operation);
console.log(`Result: ${result}`);
